var isSmartPhone = function(){
	if ((navigator.userAgent.indexOf('iPhone') > 0 && navigator.userAgent.indexOf('iPad') == -1) || navigator.userAgent.indexOf('iPod') > 0 || navigator.userAgent.indexOf('Android') > 0) {
		return true;
	}else{
		return false;
	}
}
if(isSmartPhone()){
	console.log('スマートフォン');
}else{
	console.log('スマートフォン以外');
}
